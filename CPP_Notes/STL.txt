// STL Notes: Vector, Map, Iterators

// ==================== VECTOR ====================
// Dynamic array that grows automatically
#include <vector>

std::vector<int> v;              // empty vector
std::vector<int> v2(5);          // 5 elements, initialized to 0
std::vector<int> v3(5, 10);      // 5 elements, all initialized to 10
std::vector<int> v4 = {1, 2, 3}; // initializer list

// Common operations
v.push_back(10);    // add element at end
v.pop_back();       // remove last element
v.size();           // number of elements
v.capacity();       // allocated storage size
v.empty();          // check if empty
v[0];               // access (no bounds check)
v.at(0);            // access with bounds check
v.front();          // first element
v.back();           // last element

// Modifiers
v.insert(v.begin() + 2, 50);  // insert at index 2
v.erase(v.begin() + 2);       // erase at index 2
v.clear();                    // remove all elements
v.resize(10);                 // resize to 10 elements


// ==================== MAP ====================
// Associative container with key-value pairs (sorted by key)
#include <map>

std::map<std::string, int> m;           // empty map
std::map<std::string, int> m2 = {{"a", 1}, {"b", 2}};

// Common operations
m["key"] = value;       // insert/update (O(log n))
m.insert({"key", value});           // insert pair
m.insert(std::make_pair("key", value));
m.find(key);            // returns iterator (end if not found)
m.count(key);           // returns 0 or 1
m.erase(key);           // remove by key
m.size();               // number of elements
m.empty();              // check if empty

// Access
m["key"];              // throws if key doesn't exist (inserts default)
m.at("key");           // throws if key doesn't exist

// Iteration (sorted order)
for (auto& pair : m) {
    std::cout << pair.first << ": " << pair.second << "\n";
}

// ==================== ITERATORS ====================
// General syntax: std::container<type>::iterator

// Vector iterators
std::vector<int>::iterator it = v.begin();
std::vector<int>::iterator end = v.end();

for (auto it = v.begin(); it != v.end(); ++it) {
    std::cout << *it << "\n";
}

// Range-based for (preferred)
for (int val : v) {
    std::cout << val << "\n";
}

// Map iterators
for (auto it = m.begin(); it != m.end(); ++it) {
    // it->first = key, it->second = value
    std::cout << it->first << ": " << it->second << "\n";
}

// Const iterators (read-only)
std::vector<int>::const_iterator cit = v.cbegin();

// Reverse iterators
for (auto rit = v.rbegin(); rit != v.rend(); ++rit) {
    std::cout << *rit << "\n";
}

#include <algorithm>

std::sort(v.begin(), v.end());           // sort ascending

std::find(v.begin(), v.end(), value);    // find element
